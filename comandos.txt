->configure o pc com o nome e o email do usuario
git config --global user.name "nome"
git config --global user.email "email"

->navegando no cmd
cd nome-da-pasta abre              abre pasta destino
cd ..                              volta para a pasta anterior
cd\                                vai para o c:   

->Inicializa um repositorio git (utilizado apenas 1 vez)
git init

->verificar o status do repositorio
git status

->adicionar os novos arquivos no git para versionar
git add .

->salva uma versão do arquivo
git commit -m "Porque estou salvando???"

->envia informação do computador para a nuvem
git push

->baixa a atualização ods arquivos do github pra uma pasta ja existente
git pull

->Sincroniza sem alterar a pasta local verificando mudanças no github
git fetch
